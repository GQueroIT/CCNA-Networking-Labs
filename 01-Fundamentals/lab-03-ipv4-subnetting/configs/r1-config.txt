! Router R1 Configuration
! Lab: IPv4 Subnetting
! Author: Gabriel Quero
! Description:
! This configuration demonstrates subnetting using /27 networks.
! Each interface is assigned to a different subnet to enable communication between segmented networks.

hostname R1

interface GigabitEthernet0/0
 description Subnet 1 - 192.168.10.0/27
 ip address 192.168.10.1 255.255.255.224
 no shutdown

interface GigabitEthernet0/1
 description Subnet 2 - 192.168.10.32/27
 ip address 192.168.10.33 255.255.255.224
 no shutdown

interface GigabitEthernet0/2
 description Subnet 3 - 192.168.10.64/27
 ip address 192.168.10.65 255.255.255.224
 no shutdown
